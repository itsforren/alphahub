Run automated verification against all 7 core features using the mcp-proxy endpoint. Each tool call validates that the feature works end-to-end (edge function -> database -> response).

The MCP proxy endpoint:
```
POST https://qcunascacayiiuufjtaq.supabase.co/functions/v1/mcp-proxy
Headers: Content-Type: application/json, x-mcp-secret: ${MCP_PROXY_SECRET}
Body: {"tool": "<tool_name>", "params": <params>}
```

**Run these verification calls:**

```bash
#!/bin/bash
SUPABASE_URL="https://qcunascacayiiuufjtaq.supabase.co"
MCP_SECRET="${MCP_PROXY_SECRET}"
PASS=0
FAIL=0
RESULTS=""

verify_tool() {
  local tool="$1"
  local params="${2:-{}}"
  local response

  response=$(curl -s -w "\n%{http_code}" -X POST \
    "${SUPABASE_URL}/functions/v1/mcp-proxy" \
    -H "Content-Type: application/json" \
    -H "x-mcp-secret: ${MCP_SECRET}" \
    -d "{\"tool\": \"${tool}\", \"params\": ${params}}")

  local http_code=$(echo "$response" | tail -1)
  local body=$(echo "$response" | sed '$d')

  if [ "$http_code" = "200" ] && echo "$body" | jq -e '.result' > /dev/null 2>&1; then
    echo "PASS: ${tool} (HTTP ${http_code})"
    PASS=$((PASS + 1))
    RESULTS="${RESULTS}PASS: ${tool}\n"
  else
    echo "FAIL: ${tool} (HTTP ${http_code})"
    echo "  Response: $(echo "$body" | head -c 200)"
    FAIL=$((FAIL + 1))
    RESULTS="${RESULTS}FAIL: ${tool} (HTTP ${http_code})\n"
  fi
}

echo "=== Feature Verification ==="
echo ""

# Feature 1: Client Management (CUT-05)
echo "--- Client Management ---"
verify_tool "list_clients" '{"limit": 5}'
verify_tool "search_clients" '{"query": "warren"}'
verify_tool "get_client_detail" '{"client_name": "James Warren"}'

# Feature 2: Billing (CUT-06)
echo "--- Billing ---"
verify_tool "get_billing_summary" '{}'
verify_tool "get_stripe_invoices" '{"limit": 5}'
verify_tool "get_stripe_charges" '{"limit": 5}'
verify_tool "get_stripe_subscriptions" '{"limit": 5}'

# Feature 3: Ad Spend Wallets (CUT-07)
echo "--- Ad Spend Wallets ---"
verify_tool "get_ad_spend_overview" '{}'

# Feature 4: Campaign Health (CUT-08)
echo "--- Campaign Health ---"
verify_tool "get_campaign_health" '{}'

# Feature 5: Lead Pipeline (CUT-09)
echo "--- Lead Pipeline ---"
verify_tool "get_lead_pipeline" '{}'

# Feature 6: Communications (CUT-10)
echo "--- Communications ---"
verify_tool "get_communications" '{"limit": 5}'

# Feature 7: Financial Projections & Alerts (CUT-11)
echo "--- Financial Projections & Alerts ---"
verify_tool "get_financial_projections" '{}'
verify_tool "get_alerts" '{}'

echo ""
echo "=== RESULTS ==="
echo "Passed: ${PASS}"
echo "Failed: ${FAIL}"
echo ""
if [ "$FAIL" -gt 0 ]; then
  echo "VERDICT: FAIL -- ${FAIL} tool(s) did not pass"
  echo "Fix failures before proceeding with webhook switchover"
else
  echo "VERDICT: PASS -- All ${PASS} tools verified"
fi
```

**Pass criteria:** ALL tools must return HTTP 200 with a `.result` field in the JSON response. If any tool fails:
1. Investigate the error (is it a data issue from re-sync? A function error? A secret missing?)
2. Fix the root cause
3. Re-run the failing verification

**This is a go/no-go gate.** Do NOT proceed to the checkpoint until all tools pass.
